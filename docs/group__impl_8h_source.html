141 template <typename... ReplicatedTypes>
142 template <typename FirstType, typename... RestTypes>
143 std::set<std::pair<subgroup_id_t, node_id_t>> Group<ReplicatedTypes...>::construct_objects(
144  const View& curr_view,
145  const vector_int64_2d& old_shard_leaders) {
146  std::set<std::pair<subgroup_id_t, uint32_t>> subgroups_to_receive;
147  if(!curr_view.is_adequately_provisioned) {
148  return subgroups_to_receive;
149  }
150  //The numeric type ID of this subgroup type is its position in the ordered list of subgroup types
151  const subgroup_type_id_t subgroup_type_id = index_of_type<FirstType, ReplicatedTypes...>;
152  const auto& subgroup_ids = curr_view.subgroup_ids_by_type_id.at(subgroup_type_id);
153  for(uint32_t subgroup_index = 0; subgroup_index < subgroup_ids.size(); ++subgroup_index) {
154  subgroup_id_t subgroup_id = subgroup_ids.at(subgroup_index);
155  // Find out if this node is in any shard of this subgroup
156  bool in_subgroup = false;
157  uint32_t num_shards = curr_view.subgroup_shard_views.at(subgroup_id).size();
158  for(uint32_t shard_num = 0; shard_num < num_shards; ++shard_num) {
159  const std::vector<node_id_t>& members = curr_view.subgroup_shard_views.at(subgroup_id).at(shard_num).members;
160  //"If this node is in subview->members for this shard"
161  if(std::find(members.begin(), members.end(), my_id) != members.end()) {
162  in_subgroup = true;
163  // This node may have been re-assigned from a different shard, in which
164  // case we should delete the old shard's object state
165  auto old_object = replicated_objects.template get<FirstType>().find(subgroup_index);
166  if(old_object != replicated_objects.template get<FirstType>().end() && old_object->second.get_shard_num() != shard_num) {
167  whenlog(logger->debug("Deleting old Replicated Object state for type {}; I was reassigned from shard {} to shard {}", typeid(FirstType).name(), old_object->second.get_shard_num(), shard_num));
168  replicated_objects.template get<FirstType>().erase(old_object);
169  // also erase from objects_by_subgroup_id
170  objects_by_subgroup_id.erase(subgroup_id);
171  }
172  //If we don't have a Replicated<T> for this (type, subgroup index), we just became a member of the shard
173  if(replicated_objects.template get<FirstType>().count(subgroup_index) == 0) {
174  //Determine if there is existing state for this shard that will need to be received
175  bool has_previous_leader = old_shard_leaders.size() > subgroup_id
176  && old_shard_leaders[subgroup_id].size() > shard_num
177  && old_shard_leaders[subgroup_id][shard_num] > -1
178  && old_shard_leaders[subgroup_id][shard_num] != my_id;
179  if(has_previous_leader) {
180  subgroups_to_receive.emplace(subgroup_id, old_shard_leaders[subgroup_id][shard_num]);
181  }
182  if(has_previous_leader && !has_persistent_fields<FirstType>::value) {
183  /* Construct an "empty" Replicated<T>, since all of T's state will
184  * be received from the leader and there are no logs to update */
185  replicated_objects.template get<FirstType>().emplace(
186  subgroup_index, Replicated<FirstType>(subgroup_type_id, my_id,
187  subgroup_id, subgroup_index,
188  shard_num, rpc_manager, this));
189  } else {
190  replicated_objects.template get<FirstType>().emplace(
191  subgroup_index, Replicated<FirstType>(subgroup_type_id, my_id,
192  subgroup_id, subgroup_index, shard_num, rpc_manager,
193  factories.template get<FirstType>(), this));
194  }
195  // Store a reference to the Replicated<T> just constructed
196  objects_by_subgroup_id.emplace(subgroup_id,
197  replicated_objects.template get<FirstType>().at(subgroup_index));
198  break; // This node can be in at most one shard, so stop here
199  }
200  }
201  }
202  if(!in_subgroup) {
203  // If we have a Replicated<T> for the subgroup, but we're no longer a
204  // member, delete it
205  auto old_object = replicated_objects.template get<FirstType>().find(subgroup_index);
206  if(old_object != replicated_objects.template get<FirstType>().end()) {
207  whenlog(logger->debug("Deleting old Replicated Object state (of type {}) for subgroup {} because this node is no longer a member", typeid(FirstType).name(), subgroup_index));
208  replicated_objects.template get<FirstType>().erase(old_object);
209  objects_by_subgroup_id.erase(subgroup_id);
210  }
211  // Create an ExternalCaller for the subgroup if we don't already have one
212  external_callers.template get<FirstType>().emplace(
213  subgroup_index, ExternalCaller<FirstType>(subgroup_type_id,
214  my_id, subgroup_id, rpc_manager));
215  }
216  }
217  return functional_insert(subgroups_to_receive, construct_objects<RestTypes...>(curr_view, old_shard_leaders));
218 }

286 template <typename... ReplicatedTypes>
287 void Group<ReplicatedTypes...>::receive_objects(const std::set<std::pair<subgroup_id_t, node_id_t>>& subgroups_and_leaders) {
288  //This will receive one object from each shard leader in ascending order of subgroup ID
289  for(const auto& subgroup_and_leader : subgroups_and_leaders) {
290  LockedReference<std::unique_lock<std::mutex>, tcp::socket> leader_socket
291  = tcp_sockets->get_socket(subgroup_and_leader.second);
292  ReplicatedObject& subgroup_object = objects_by_subgroup_id.at(subgroup_and_leader.first);
293  if(subgroup_object.is_persistent()) {
294  int64_t log_tail_length = subgroup_object.get_minimum_latest_persisted_version();
295  whenlog(logger->debug("Sending log tail length of {} for subgroup {} to node {}.", log_tail_length, subgroup_and_leader.first, subgroup_and_leader.second));
296  leader_socket.get().write(log_tail_length);
297  }
298  whenlog(logger->debug("Receiving Replicated Object state for subgroup {} from node {}", subgroup_and_leader.first, subgroup_and_leader.second));
299  std::size_t buffer_size;
300  bool success = leader_socket.get().read(buffer_size);
301  assert_always(success);
302  char* buffer = new char[buffer_size];
303  success = leader_socket.get().read(buffer, buffer_size);
304  assert_always(success);
305  subgroup_object.receive_object(buffer);
306  delete[] buffer;
307  }
308  whenlog(logger->debug("Done receiving all Replicated Objects from subgroup leaders"));
309 }
